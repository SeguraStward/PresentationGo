Structs: Composición de Tipos

Definición y Uso

type Persona struct {
    Nombre   string
    Apellido string
    Edad     int
}

// Creación
p := Persona{
    Nombre: "Ana",
    Apellido: "López",
    Edad: 30,
}

// Acceso
fmt.Println(p.Nombre)

Embedding (Composición)

type Empleado struct {
    Persona  // Embedding
    ID       string
    Salario  float64
}

e := Empleado{
    Persona: Persona{
        Nombre: "Carlos",
    },
    ID: "E123",
}

// Campos promocionados
fmt.Println(e.Nombre)

Métodos e Interfaces

Métodos con Receptores

// Receptor de valor
func (p Persona) NombreCompleto() string {
    return p.Nombre + " " + p.Apellido
}

// Receptor de puntero (modifica)
func (p *Persona) CumplirAnios() {
    p.Edad++
}

p := Persona{Nombre: "Ana", Edad: 25}
p.CumplirAnios()  // Edad ahora es 26

Interfaces Implícitas

type Saludable interface {
    Saludar() string
}

func (p Persona) Saludar() string {
    return "Hola, " + p.Nombre
}

// Polimorfismo
func SaludarA(s Saludable) {
    fmt.Println(s.Saludar())
}

SaludarA(p)  // Funciona automáticamente

Concurrencia: Goroutines y Channels

Goroutines: Hilos Ligeros

// Lanzar goroutine
go func() {
    fmt.Println("Ejecutando concurrentemente")
}()

// Con WaitGroup
var wg sync.WaitGroup
wg.Add(1)
go func() {
    defer wg.Done()
    // hacer trabajo
}()
wg.Wait()

Channels: Comunicación Segura

// Canal sin buffer
ch := make(chan int)

go func() {
    ch <- 42  // Enviar
}()

valor := <-ch  // Recibir

// Canal con buffer
buf := make(chan int, 3)
buf <- 1
buf <- 2
buf <- 3

Manejo de Errores Explícito

Patrón Estándar

func dividir(a, b float64) (float64, error) {
    if b == 0 {
        return 0, errors.New("división por cero")
    }
    return a / b, nil
}

// Uso
resultado, err := dividir(10, 0)
if err != nil {
    fmt.Println("Error:", err)
    return
}
fmt.Println(resultado)
